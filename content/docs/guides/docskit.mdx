### Marks

Highlighting lines with default and custom colors:

```js
function lorem(ipsum, dolor = 1) {
  // !mark
  return dolor;
}

function ipsum(lorem, dolor = 1) {
  // !mark(1:2) gold
  const sit = lorem == null ? 0 : 1;
  dolor = sit - amet(dolor);
  return sit ? consectetur(lorem) : [];
}
```

Highlighting tokens using regex:

```js
// !mark[/sit/mg] #d946ef
function ipsum(lorem, dolor = 1) {
  // !mark[/lorem == null/]
  const sit = lorem == null ? 0 : 1;
  dolor = sit - amet(dolor);
  return sit ? consectetur(lorem) : [];
}
```

Highlighting diffs:

```js
function lorem(ipsum, dolor = 1) {
  const sit = 10;
  // !diff -
  dolor = ipsum - sit;
  // !diff +
  dolor = sit - amet(dolor);
  return sit ? consectetur(ipsum) : [];
}
```
