<API>

<APIInfo method={"GET"} route={"/runes/v1/addresses/{address}/balances"}>

Retrieves a paginated list of address balances.

### Query Parameters

<Property required={false} deprecated={false} name={"offset"} type={"integer"}>

Result offset

<span>Minimum:  `0`</span>

</Property>

<Property required={false} deprecated={false} name={"limit"} type={"integer"}>

Results per page

<span>Minimum:  `1`</span>

<span>Maximum:  `60`</span>

</Property>

### Path Parameters

<Property required={true} deprecated={false} name={"address"} type={"string"}>

Bitcoin address

<span>Example:  `bc1p8aq8s3z9xl87e74twfk93mljxq6alv4a79yheadx33t9np4g2wkqqt8kc5`</span>

</Property>

| Status code | Description |
| :----------- | :----------- |
| `200` | Default Response |

</APIInfo>

<APIExample>

```bash title="curl"
curl -X GET "https://api.hiro.so/runes/v1/addresses/bc1q7jd477wc5s88hsvenr0ddtatsw282hfjzg59wz/balances"
```

<Accordion title={"Typescript Definition"}>

```ts
/**
 * Rune ID
 */
export type ID = string;
/**
 * Rune name
 */
export type Name = string;
/**
 * Rune name with spacers
 */
export type SpacedName = string;
/**
 * Bitcoin address which holds this balance
 */
export type Address = string;
/**
 * Rune balance
 */
export type Balance = string;

export interface PaginatedBalancesResponse {
  limit: number;
  offset: number;
  total: number;
  results: ({
    rune: RuneDetail;
  } & {
    address?: Address;
    balance: Balance;
  })[];
}
/**
 * Details of the rune affected by this activity
 */
export interface RuneDetail {
  id: ID;
  name: Name;
  spaced_name: SpacedName;
}

```
