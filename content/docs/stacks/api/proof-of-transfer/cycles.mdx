<APIInfo method={"GET"} route={"/extended/v2/pox/cycles"}>

Retrieves a list of PoX cycles.

### Query parameters

<Property required={false} deprecated={false} name={"limit"} type={"integer"}>

max number of cycles to fetch

<span>Default:  `20`</span>

<span>Maximum:  `60`</span>

</Property>

<Property required={false} deprecated={false} name={"offset"} type={"integer"}>

index of first cycle to fetch

<span>Example:  `20`</span>

</Property>

| Status code | Description |
| :--------- | :---------- |
| `200` | List of cycles |

</APIInfo>

<APIExample>

```bash title="Terminal"
curl -X GET "https://api.mainnet.hiro.so/extended/v2/pox/cycles"
```

/**
 * GET request that returns PoX cycles
 */
export interface PoxCycleListResponse {
  /**
   * The number of cycles to return
   */
  limit: number;
  /**
   * The number to cycles to skip (starting at `0`)
   */
  offset: number;
  /**
   * The total number of cycles
   */
  total: number;
  results: PoxCycle[];
}
export interface PoxCycle {
  block_height: number;
  index_block_hash: string;
  cycle_number: number;
  total_weight: number;
  total_stacked_amount: string;
  total_signers: number;
}
